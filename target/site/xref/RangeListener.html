1   /**
2    * The "model" doesn't know the specific "GUI" classes.
3    * But it still wants to notify the "GUI" (or text UI, or ...)
4    * whenever the "model" changes, so that the "GUI" (or text UI, or ...)
5    * can react to those changes (e.g., by repainting itself).
6    * 
7    * <p>To enable this, we have this RangeListener interface.
8    * The "model" (i.e., the Range) will call rangeChanged() 
9    * on all registered listeners,
10   * without knowing the specific subtype of RangeListener
11   * (e.g., the "model" does not know about the PlotCanvas class).
12   */
13  public interface RangeListener {
14  
15      /**
16       * Notify this RangeListener that the given Range has changed.
17       * @param range The Range that changed.
18       */
19      public abstract void rangeChanged(Range range);
20      
21  }
