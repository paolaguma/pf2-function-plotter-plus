1   import java.awt.Color;
2   import java.awt.Dimension;
3   import java.awt.Graphics;
4   import java.awt.Graphics2D;
5   import java.awt.RenderingHints;
6   import java.awt.event.MouseAdapter;
7   import java.awt.event.MouseEvent;
8   import java.awt.event.MouseMotionAdapter;
9   
10  /**
11   * The PlotCanvas is part of the GUI.
12   * It is a custom component, which draws itself
13   * (in the paintComponent method)
14   * and which provides information on how big it would like to be
15   * (via the getPreferredSize method).
16   */
17  public final class PlotCanvas extends Canvas {
18  
19      private static final Dimension PREFERRED_SIZE = new Dimension(400, 300);
20      
21      private final Cursor cursor;
22      
23      
24      /**
25       * Create a PlotCanvas for the given Plot.
26       * @param plot The Plot to show
27       */
28      public PlotCanvas(final Plot plot) {
29          super(plot);
30          this.cursor = plot.getCursor();
31          
32          // register listeners        
33          cursor.addCursorListener(new CursorListener() {
34              public void cursorChanged(final Cursor cursor) {
35                  repaint();
36              }
37          });
38          addMouseListener(new MouseAdapter() {
39              public void mousePressed(final MouseEvent ev) {
40                  cursor.setX(xViewToModel(ev.getX()));
41              }
42          });
43          addMouseMotionListener(new MouseMotionAdapter() {
44              public void mouseDragged(final MouseEvent ev) {
45                  cursor.setX(xViewToModel(ev.getX()));
46              }
47          });
48      }
49      
50      @Override
51      public Dimension getPreferredSize() {
52          return PREFERRED_SIZE;
53      }
54      
55      @Override
56      public void paintComponent(final Graphics g) {
57          // fill the background with white
58          g.setColor(Color.WHITE);
59          g.fillRect(0, 0, getWidth(), getHeight());
60          
61          // find min and max model y
62          final double minModelY = range.getMin();
63          final double maxModelY = range.getMax();
64  
65          drawAxes(g, minModelY, maxModelY);
66  
67          // plot curve
68          drawFunction(g, minModelY, maxModelY);
69          
70          // draw x and y cursor
71          drawCursors(g, minModelY, maxModelY);
72      }   
73  
74      private void drawFunction(final Graphics g, final double minModelY, final double maxModelY) {
75          g.setColor(Color.BLUE);
76          //iterate through vx(viewX) -> xViewtoModel
77          // calculate yModel using function.f(x);
78          // plot y to view yModeltoView
79          final Graphics2D g2d = (Graphics2D)g;
80          int prevX = 0;
81          int prevY = 0;
82          for (int vx = 0; vx < getWidth(); vx++) {
83              
84              final double xModel = xViewToModel(vx);
85              final double yModel = function.compute(xModel);
86              final int vy = yModelToView(yModel, minModelY, maxModelY);
87              
88              if (vx == 0) {
89                  prevX = vx;
90                  prevY = vy;
91              }
92              g2d.setRenderingHint(
93                  RenderingHints.KEY_ANTIALIASING, 
94                  RenderingHints.VALUE_ANTIALIAS_ON);
95              g2d.drawLine(prevX, prevY, vx, vy);
96  
97              prevX = vx;
98              prevY = vy;
99          }
100     }
101 
102     private void drawAxes(final Graphics g, final double minModelY, final double maxModelY) {
103         // plot x axis (at y=0)
104         g.setColor(Color.GRAY);
105         final int viewY0 = yModelToView(0, minModelY, maxModelY);
106         g.drawLine(0, viewY0, getWidth(), viewY0);
107         
108         // plot y axis (x = 0)
109         final int viewX0 = xModelToView(0);
110         g.drawLine(viewX0, 0, viewX0, getHeight());
111     }
112 
113     private void drawCursors(final Graphics g, final double minModelY, final double maxModelY) {
114         g.setColor(Color.BLACK);
115         final int cursorViewX = xModelToView(cursor.getX());
116         g.drawLine(cursorViewX, 0, cursorViewX, getHeight());
117         g.drawString("x = " + cursor.getX(), cursorViewX + 4, getHeight() - 20);
118         // draw y cursor
119         final double cursorModelY = function.compute(cursor.getX());
120         final int cursorViewY = yModelToView(cursorModelY, minModelY, maxModelY);
121         g.drawLine(0, cursorViewY, getWidth(), cursorViewY);
122         g.drawString("y = " + cursorModelY, cursorViewX + 4, getHeight() - 4);
123     }
124     
125     private double xViewToModel(final int vx) {
126         return ((double)vx) / getWidth() * range.getExtent() + range.getMin();
127     }
128     
129     private int xModelToView(final double x) {
130         return (int)((x - range.getMin()) / range.getExtent() * getWidth());
131 
132     }
133     
134     private int yModelToView(final double y, final double minY, final double maxY) {
135         return (int)(getHeight() - 1 - (y - minY) / (maxY - minY) * getHeight());
136     }
137 }
