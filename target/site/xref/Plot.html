1   /**
2    * The Plot is the "model" of this application.
3    * It is an immutable class that
4    * points to the two mutable and observable parts of the "model":
5    * 1) a Function, and
6    * 2) a Range (along the x-axis, over which the function should be plotted).
7    */
8   public final class Plot {
9   
10      private final Function function;
11      private final Range range;
12      private final Cursor cursor;
13      
14      
15      /**
16       * Create a new Plot of the given Function, with the given Range and Cursor.
17       * @param function The Function to plot
18       * @param range The Range (on the x-axis) over which to plot
19       * @param cursor The Cursor denoting a point on the x-axis
20       */
21      public Plot(final Function function, final Range range, final Cursor cursor) {
22          this.function = function;
23          this.range = range;
24          this.cursor = cursor;
25      }
26      
27      /**
28       * Get the Function of this Plot.
29       * @return this Plot's Function
30       */
31      public final Function getFunction() {
32          return function;
33      }
34      
35      /**
36       * Get the Range of this Plot.
37       * @return this Plot's Range
38       */
39      public final Range getRange() {
40          return range;
41      }
42      
43      /**
44       * Get the Cursor of this Plot.
45       * @return this Plot's Cursor
46       */
47      public final Cursor getCursor() {
48          return cursor;
49      }
50      
51  }
