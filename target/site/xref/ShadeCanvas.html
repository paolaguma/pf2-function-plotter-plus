7   /**
8    * The ShadeCanvas is part of the GUI.
9    * It is a custom component, which draws itself
10   * (in the paintComponent method)
11   * and which provides information on how big it would like to be
12   * (via the getPreferredSize method).
13   */
14  public final class ShadeCanvas extends Canvas {
15  
16      private static final Dimension PREFERRED_SIZE = new Dimension(400, 20);
17      
18      
19      /**
20       * Create a new ShadeCanvas presenting the given Plot.
21       * @param plot The Plot to render
22       */
23      public ShadeCanvas(final Plot plot) {
24          super(plot);
25      }
26      
27      @Override
28      public final Dimension getPreferredSize() {
29          return PREFERRED_SIZE;
30      }
31      
32      @Override
33      public final void paintComponent(final Graphics g) {
34          g.setColor(Color.WHITE);
35          g.fillRect(0, 0, getWidth(), getHeight());
36          
37          final Double minModelY = range.getMin();
38          final Double maxModelY = range.getMax();
39  
40          for (int vx = 0; vx < getWidth(); vx ++) {
41  
42              final double xModel = xViewToModel(vx);
43              final double yModel = function.compute(xModel);
44              final int vy = yModelToView(yModel, minModelY, maxModelY);
45  
46              int greyValue;
47              // keeping the range from 0 to 255
48              if (vy < 0) {
49                  greyValue = 0; 
50              } else if (vy >= 255) {
51                  greyValue = 255;
52              } else {
53                  greyValue = vy; 
54              }
55  
56              // inversing the colors.
57              greyValue = 255 - greyValue;
58              final Color c = new Color(greyValue, greyValue, greyValue);
59              
60              g.setColor(c);
61              g.drawLine(vx, 0, vx, getHeight()); 
62          }
63  
64      }
65  
66      /**
67       * Convert the given view x-coordinate into a model coordinate.
68       * @param vx an x-coordinate in the view (pixels from left of canvas)
69       * @return x-coordinate in the model
70       */
71      private double xViewToModel(final int vx) {
72          return ((double)vx) / getWidth() * range.getExtent() + range.getMin();
73      }
74      
75      /**
76       * Convert the given model y-coordinate into a view coordinate.
77       * @param y an y-coordinate in the model
78       * @return y-coordinate in the view (pixels from top of canvas)
79       */
80      private int yModelToView(final double y, final double minY, final double maxY) {
81          return (int)((y - minY) / (maxY - minY) * 255);
82      }
83      
84  }
